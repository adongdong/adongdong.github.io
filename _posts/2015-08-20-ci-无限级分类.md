    public function sel_dg($arr,$parent_id=0,$level=0){
            static $data = array();
            foreach($arr as $k=>$v) {
                if($v['parent_id']==$parent_id){
                    $v['level']=$level;
                    $data[]=$v;
                    $this->sel_dg($arr,$v['cat_id'],$level+1);
                }
            }
            return $data;
        }
